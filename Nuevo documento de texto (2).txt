Funcion vidaEnemigo = attack(dmg, vidaEnemigo, vidaMaximaEnem, trueDmg)
	Definir c Como Real;
	c = 0;
	trueDmg[0] = dmg;
	vidaEnemigo = vidaEnemigo - dmg;
	c = vidaEnemigo;
	si vidaEnemigo < 0 Entonces
		vidaEnemigo = 0;
		trueDmg[0] = trueDmg[0] - abs(c);
	FinSi
	
	
FinFuncion

Funcion vida = healing(heal, vida, vidaMaxima, trueHeal)
	
	trueHeal[0] = heal;
	vida = vida + heal;
	si vida > vidaMaxima Entonces
		trueHeal[0] = heal - (vida - vidaMaxima);
		vida = vidaMaxima;
	FinSi
	
	
FinFuncion

Funcion d = damageIncrease(d, aumentador, aumento)
	Segun aumento[0] Hacer
		
		"*": d = d * aumentador[0];
			
		"^": d = d ^ aumentador[0];
			
		"+": d = d + aumentador[0];
			
		
	FinSegun
	
FinFuncion

Funcion costoActual = extraAttac(costoActual, atacarExtra)
	costoActual = costoActual / atacarExtra;
FinFuncion

Funcion vidaMaxima = aumentarVida(vidaMaxima, vidaAumentada)
	vidaMaxima = vidaMaxima * vidaAumentada;
FinFuncion

Funcion dmgEnemy = Defensa(dmgEnemy, def)
	dmgEnemy = dmgEnemy - (def * (def)/100);
FinFuncion

Funcion dmg = DefensaEnemigo(dmg, defEnemy)
	dmg = dmg - (dmg * (defEnemy)/100);
FinFuncion

Funcion def = defff(def, masDefensa, menDef)
	def = masDefensa - menDef;
	si def > 100 Entonces
		def = 100;
	FinSi
	
FinFuncion
Funcion defEnemy = deffE(defEnemy, masDefensaEnem, menDefEnem)
	defEnemy = masDefensaEnem - menDefEnem;
	si defEnemy > 100 Entonces
		defEnemy = 100;
	FinSi
	
FinFuncion

FinFuncion

Funcion mostrarAumento(aumentador, durAumentador, aumento, damage, a, c, d)

FinFuncion

Funcion redondearVida(vida, vidaMaxima, VidaEnemigo, vidaMaximaEnem, v, vE)
	Definir ax, axVm Como Entero;
	ax = trunc(vida);
	axVm = trunc(vidaMaxima);
	si axVm < vidaMaxima Entonces
		axVm = axVm + 1;
	FinSi
	si vida == vidaMaxima Entonces
		v[0] = axVm;
	SiNo
		si vida <= 0 Entonces
			v[0] = 0;
		SiNo
			si vida > 0  | vida < vidaMaxima Entonces
				v[0] = ax + 1;
			FinSi
		FinSi
		
	FinSi
	
	ax = trunc(vidaEnemigo);
	axVm = trunc(vidaMaximaEnem);
	si axVm < vidaMaximaEnem Entonces
		axVm = axVm + 1;
	FinSi
	si vidaEnemigo == vidaMaximaEnem Entonces
		vE[0] = axVm;
	SiNo
		si vidaEnemigo <= 0 Entonces
			vE[0] = 0;
